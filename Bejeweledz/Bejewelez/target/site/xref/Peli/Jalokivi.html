6   package Peli;

8   import java.util.List;
9   import java.util.Map;
10  import java.util.Random;
11  import java.util.stream.Collectors;
12  import javafx.beans.property.IntegerProperty;
13  import javafx.beans.property.SimpleIntegerProperty;
14  import javafx.event.EventHandler;
15  import javafx.geometry.Point2D;
16  import javafx.scene.Parent;
17  import javafx.scene.input.MouseEvent;
18  import javafx.scene.paint.Color;
19  import javafx.scene.paint.Paint;
20  import javafx.scene.shape.Circle;
21  
22  /**
23   *
24   * @author mkatri
25   */
26  public class Jalokivi extends Parent {
27  /* Samat muuttujat, kun Pelipaneelissa */
28      private static final int Koko = 100;
29      private static final int Leveys = 6;
30      private static final int Korkeus = 6;
31  
32      private Color[] varit = new Color[]{Color.AQUA, Color.DEEPPINK, Color.BLUEVIOLET, Color.CORAL, Color.YELLOW, Color.CRIMSON};
33  
34      private Jalokivi selected = null;
35      private List<Jalokivi> jalokivet;
36  
37      private IntegerProperty pisteet = new SimpleIntegerProperty();
38  
39      private Circle ympyra = new Circle(Koko / 2);
40      
41  /*MouseEvent vielä epäselvä, mutta pitäisi toimia, kun luokat saa yhdistettyä, eli jalokivien liikkuminen
42      tapahtuu hiiren klikkauksella ja jokaista jalokiveä pystyy siirtämään minkä vaan jalokiven päälle.
43      Jos rivin saa täyteen samoja jalokiviä, siirrytään tarkistaJalokivet() metodiin*/
44      public Jalokivi(Point2D point) {
45          
46          ympyra.setCenterX(Koko / 2);
47          ympyra.setCenterY(Koko / 2);
48          ympyra.setFill(varit[new Random().nextInt(varit.length)]);
49  
50          setTranslateX(point.getX() * Koko);
51          setTranslateY(point.getY() * Koko);
52          getChildren().add(ympyra);
53  
54          setOnMouseClicked(new EventHandler<MouseEvent>() {
55              @Override
56              public void handle(MouseEvent event) {
57                  if (selected == null) {
58                      selected = Jalokivi.this;
59                  } else {
60                      vaihdaJalokivet(selected, Jalokivi.this);
61                      tarkistaJalokivet();
62                      selected = null;
63                  }
64              }
65          });
66      }
67  /*Lisätään jalokiville randomisti väri kuudesta eri värivaihtoehdosta */
68      public void randomi() {
69          ympyra.setFill(varit[new Random().nextInt(varit.length)]);
70      }
71  
72      public int getSarake() {
73          return (int) getTranslateX() / Koko;
74      }
75  
76      public int getRivi() {
77          return (int) getTranslateY() / Koko;
78      }
79  
80      public void setVari(Paint color) {
81          ympyra.setFill(color);
82      }
83  
84      public Paint getVari() {
85          return ympyra.getFill();
86      }
87  /*tarkistetaan jalokivet rivi ja sarake kerrallaan */
88      public void tarkistaJalokivet() {
89          Map<Integer, List<Jalokivi>> rows = jalokivet.stream().collect(Collectors.groupingBy(Jalokivi::getRivi));
90          Map<Integer, List<Jalokivi>> columns = jalokivet.stream().collect(Collectors.groupingBy(Jalokivi::getSarake));
91  
92          rows.values().forEach(this::tarkistaYhdistelmat);
93          columns.values().forEach(this::tarkistaYhdistelmat);
94      }
95  /* katsotaan, onko jalokivistä yhdistelmiä, ja jos on, niin riville otetaan randomilla uudet jalokivet ja pisteisiin 
96      lisätään 100
97      */
98      public void tarkistaYhdistelmat(List<Jalokivi> jewelsLine) {
99  
100         Jalokivi jewel = jewelsLine.get(0);
101         long count = jewelsLine.stream().filter(j -> j.getVari() != jewel.getVari()).count();
102         if (count == 0) {
103             pisteet.set(pisteet.get() + 100);
104             jewelsLine.forEach(Jalokivi::randomi);
105         }
106     }
107 /*hiirenklikkauksella vaihdetaan jalokivien paikkaa paikasta a, paikkaan b */
108     public void vaihdaJalokivet(Jalokivi a, Jalokivi b) {
109         Paint color = a.getVari();
110         a.setVari(b.getVari());
111         b.setVari(color);
112     }
